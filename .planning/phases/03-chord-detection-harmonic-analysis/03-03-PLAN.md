       ```typescript

       for (let ch = 0; ch < buffer.numberOfChannels; ch++) {
         const data = buffer.getChannelData(ch);
         const gain = 1 / buffer.numberOfChannels;
         for (let i = 0; i < buffer.length; i++) mono[i] += data[i] * gain;
       }
       ```
    6. For each second:
       ```typescript
       for (let sec = 0; sec < numSeconds; sec++) {
         const startSample = Math.floor(sec * sampleRate);
         const slice = mono.subarray(startSample, startSample + FRAME_SIZE);
         if (slice.length < FRAME_SIZE) { tension[sec] = 0; continue; }

         const chromaRaw = Meyda.extract('chroma', slice) as number[];
         if (!chromaRaw) { tension[sec] = 0; continue; }

         // Cosine similarity match (inline for offline -- no state dependency)
         let best1Score = -1, best2Score = -1, best1Idx = 0;
         for (let i = 0; i < CHORD_TEMPLATES.length; i++) {
           const tmpl = CHORD_TEMPLATES[i].template;
           let dot = 0, magA = 0, magB = 0;
           for (let j = 0; j < 12; j++) {
             dot  += chromaRaw[j] * tmpl[j];
             magA += chromaRaw[j] * chromaRaw[j];
             magB += tmpl[j] * tmpl[j];
           }
           const denom = Math.sqrt(magA) * Math.sqrt(magB);
           const s = denom === 0 ? 0 : dot / denom;
           if (s > best1Score) { best2Score = best1Score; best1Score = s; best1Idx = i; }
           else if (s > best2Score) { best2Score = s; }
         }

         const fn = CHORD_TEMPLATES[best1Idx].function;
         tension[sec] = TENSION_MIDPOINTS[fn];
       }
       ```

    8. Return tension (no normalization needed -- values are already in [0, 1] from TENSION_MIDPOINTS)
