  <action>
    Create src/components/ChordDisplay.tsx:

    ```typescript
    import { useAppStore } from '../store/useAppStore';

    const CONFIDENCE_STYLES: Record<'low' | 'medium' | 'high', { bg: string; text: string; label: string }> = {
      low:    { bg: 'rgba(107, 114, 128, 0.3)', text: '#9ca3af', label: 'low' },
      medium: { bg: 'rgba(245, 158, 11, 0.2)',  text: '#f59e0b', label: 'med' },
      high:   { bg: 'rgba(34, 197, 94, 0.2)',   text: '#22c55e', label: 'high' },
    };

    export function ChordDisplay() {
      const currentChord = useAppStore((s) => s.currentChord);
      const chordConfidence = useAppStore((s) => s.chordConfidence);
      const chordFunction = useAppStore((s) => s.chordFunction);
      const currentTension = useAppStore((s) => s.currentTension);

      const conf = CONFIDENCE_STYLES[chordConfidence];

      return (
        <div
          className="flex items-center gap-4 px-6 py-3 rounded-lg w-full"
          style={{ backgroundColor: '#13131f', border: '1px solid rgba(99,102,241,0.3)' }}
        >
          {/* Chord name */}
          <div className="flex flex-col flex-1 min-w-0">
            <div className="flex items-center gap-3">
              <span
                className="text-2xl font-bold font-mono tracking-wide truncate"
                style={{ color: '#e0e7ff' }}
              >
                {currentChord}
              </span>

              {/* Confidence badge */}
              {currentChord !== '--' && (
                <span
                  className="text-xs font-medium px-2 py-0.5 rounded-full uppercase tracking-wider"
                  style={{ backgroundColor: conf.bg, color: conf.text }}
                >
                  {conf.label}
                </span>
              )}
            </div>

            {/* Function label */}
            {chordFunction && (
              <span
                className="text-xs font-medium mt-1 truncate"
                style={{ color: '#a78bfa' }}
              >
                {chordFunction}
              </span>
            )}
          </div>

          {/* Tension readout (compact number) */}
          <div className="flex flex-col items-end">
            <span
              className="text-xs font-medium uppercase tracking-wider"
              style={{ color: '#6b7280' }}
            >
              tension
            </span>
            <span
              className="text-lg font-mono font-bold"
              style={{
                color: currentTension < 0.3 ? '#3b82f6'
                     : currentTension < 0.6 ? '#f59e0b'
                     : currentTension < 0.8 ? '#f97316'
                     : '#ef4444'
              }}
            >
              {currentTension.toFixed(2)}
            </span>
          </div>
        </div>
      );
    }
    ```

    File is straightforward -- reads from Zustand, renders chord name + confidence badge + function label + tension number. All styling uses inline styles and Tailwind classes consistent with existing components (FileUpload, InstrumentRoleOverlay).
