// CallResponseEntry: a single call-and-response event (Phase 8)
export interface CallResponseEntry {
  callSec: number;       // audioCtx.currentTime when keyboard went melodic
  responseSec: number;   // audioCtx.currentTime when guitar responded
  gapSec: number;        // responseSec - callSec
}

// CallResponseState: sliding window state for call-response detection (Phase 8)
export interface CallResponseState {
  lastKbMelodicSec: number;          // -1 = no active call
  lastDetectedResponseSec: number;   // debounce: don't log same response twice
}

**Create src/audio/CallResponseDetector.ts:**

```typescript
// Factory: initCallResponseState(): CallResponseState
//   returns { lastKbMelodicSec: -1, lastDetectedResponseSec: -1 }

// Main function: updateCallResponse(
//   crState: CallResponseState,
//   kbIsMelodic: boolean,
//   gtIsMelodic: boolean,
//   audioTimeSec: number
// ): CallResponseEntry | null
//
// Logic:
// 1. If keyboard isMelodic and lastKbMelodicSec === -1:
//    → Record lastKbMelodicSec = audioTimeSec (start of call)
// 2. If keyboard is NOT melodic and lastKbMelodicSec > 0:
//    → Keep the recorded time (call ended, waiting for response)
// 3. If guitar isMelodic and lastKbMelodicSec > 0:
//    gap = audioTimeSec - lastKbMelodicSec
//    if (gap >= 2.0 && gap <= 4.0 && audioTimeSec !== lastDetectedResponseSec):
//      → Create CallResponseEntry { callSec: lastKbMelodicSec, responseSec: audioTimeSec, gapSec: gap }
//      → Set lastDetectedResponseSec = audioTimeSec
//      → Reset lastKbMelodicSec = -1 (prevent cascading)
//      → Return the entry
// 4. If lastKbMelodicSec > 0 and (audioTimeSec - lastKbMelodicSec) > 4.0:
//    → Expired call — reset lastKbMelodicSec = -1
// 5. Return null (no event this tick)
```
