    **Design note on frequency band choice (BEAT-01):** The spec mentions snare 200-800Hz (fundamental), but we use drums_high (2000-8000Hz) instead. Reason: The 200-800Hz range overlaps heavily with bass guitar, piano comping, and vocal fundamentals in jazz — using it for onset detection would produce massive false-positive rates. The snare *crack* (broadband transient energy) is far more prominent and separable in the 2-8kHz range. The drums_high band is already defined in FrequencyBandSplitter.ts and validated in Phase 2. The ride band (6-10kHz) captures cymbal ping transients. Together, these high-frequency bands provide reliable drum onset detection without bass/harmony bleed.

    **computeDrumFlux (exported):**

    ```typescript
    export function computeDrumFlux(
      rawFreqData: Uint8Array,
      prevRawFreqData: Uint8Array,
      drumsHighBand: FrequencyBand,
      rideBand: FrequencyBand,
    ): number {
      let flux = 0;
      // Snare crack / hi-hat body (drums_high: 2000-8000 Hz)
      for (let i = drumsHighBand.lowBin; i <= drumsHighBand.highBin; i++) {
        const diff = rawFreqData[i] - prevRawFreqData[i];
        if (diff > 0) flux += diff;  // half-wave rectification
      }
      // Ride cymbal ping (ride: 6000-10000 Hz)
      for (let i = rideBand.lowBin; i <= rideBand.highBin; i++) {
        const diff = rawFreqData[i] - prevRawFreqData[i];
        if (diff > 0) flux += diff;
      }
      return flux;
    }
    ```

    **adaptiveThreshold (exported):**
    Compute mean + DRUM_THRESHOLD_MULTIPLIER * stddev over the rolling buffer.
    ```typescript
    export function adaptiveThreshold(buffer: Float32Array, samples: number): number {
      if (samples < 3) return Infinity;  // too few samples, suppress onsets
      const n = Math.min(samples, buffer.length);
      let sum = 0;
      for (let i = 0; i < n; i++) sum += buffer[i];
      const mean = sum / n;
      let variance = 0;
      for (let i = 0; i < n; i++) {
        const d = buffer[i] - mean;
        variance += d * d;
      }
      const stddev = Math.sqrt(variance / n);
      return mean + DRUM_THRESHOLD_MULTIPLIER * stddev;
    }
    ```

    **detectDrumOnset (exported):**
    Main per-tick function for drum onset detection.
    ```typescript
    export function detectDrumOnset(
      beat: BeatState,
      rawFreqData: Uint8Array,
      prevRawFreqData: Uint8Array,
      drumsHighBand: FrequencyBand,
      rideBand: FrequencyBand,
      audioTimeSec: number,
    ): boolean {
      // 1. Compute band-limited drum flux
      const flux = computeDrumFlux(rawFreqData, prevRawFreqData, drumsHighBand, rideBand);

      // 2. Push flux into rolling buffer for adaptive threshold
      beat.drumFluxBuffer[beat.drumFluxHead] = flux;
      beat.drumFluxHead = (beat.drumFluxHead + 1) % FLUX_WINDOW;
      if (beat.drumFluxSamples < FLUX_WINDOW) beat.drumFluxSamples++;

      // 3. Push flux into OSS ring buffer (for autocorrelation in BpmTracker)
      beat.ossBuffer[beat.ossHead] = flux;
      beat.ossHead = (beat.ossHead + 1) % OSS_CAP;
      if (beat.ossSamples < OSS_CAP) beat.ossSamples++;

      // 4. Compute adaptive threshold
      const threshold = adaptiveThreshold(beat.drumFluxBuffer, beat.drumFluxSamples);

      // 5. Onset fires when: flux > threshold AND flux > previous flux (rising edge)
      const isOnset = flux > threshold && flux > beat.prevDrumFlux;

      // 6. Store previous flux for next tick's rising-edge check
      beat.prevDrumFlux = flux;

      if (isOnset) {
        // Record onset timestamp in ring buffer
        beat.drumOnsetTimes[beat.drumOnsetHead] = audioTimeSec;
        beat.drumOnsetHead = (beat.drumOnsetHead + 1) % ONSET_CAP;
        if (beat.drumOnsetCount < ONSET_CAP) beat.drumOnsetCount++;

        // Update last drum onset time
        beat.lastDrumOnsetSec = audioTimeSec;

        // Increment beat counter (mod 4 for downbeat detection)
        beat.beatCounter = (beat.beatCounter + 1) % 4;
        if (beat.beatCounter === 0) {
          beat.lastDownbeatSec = audioTimeSec;
        }
      }

      return isOnset;
    }
    ```
