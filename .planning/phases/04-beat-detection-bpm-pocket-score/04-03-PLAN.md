    **computeIoiCV (exported):**
    Computes coefficient of variation from drum onset timestamps. Uses pre-allocated ioiBuffer on BeatState.
    ```typescript
    export function computeIoiCV(beat: BeatState): number {
      const count = beat.drumOnsetCount;
      if (count < 4) return 1.0;  // too few onsets → assume rubato

      // Compute IOIs from drum onset ring buffer
      // Read onsets in chronological order
      const n = Math.min(count, ONSET_CAP);
      const startIdx = count <= ONSET_CAP
        ? 0
        : beat.drumOnsetHead;  // oldest entry when buffer has wrapped

      let ioiCount = 0;
      let sum = 0;

      for (let i = 0; i < n - 1; i++) {
        const idx1 = (startIdx + i) % ONSET_CAP;
        const idx2 = (startIdx + i + 1) % ONSET_CAP;
        const ioi = beat.drumOnsetTimes[idx2] - beat.drumOnsetTimes[idx1];

        // Skip negative IOIs (can happen at ring buffer wrap with unsorted times)
        if (ioi <= 0) continue;

        beat.ioiBuffer[ioiCount] = ioi;
        sum += ioi;
        ioiCount++;
      }

      if (ioiCount < 3) return 1.0;  // too few valid IOIs

      const mean = sum / ioiCount;
      let variance = 0;
      for (let i = 0; i < ioiCount; i++) {
        const d = beat.ioiBuffer[i] - mean;
        variance += d * d;
      }
      const cv = Math.sqrt(variance / ioiCount) / mean;
      return cv;
    }
    ```

Sync score computed within a +-80ms window: score = 1 - (|offset| / 80) for offsets within window, 0 otherwise. Rolling 8-beat average provides stable pocket reading. Timing offset (positive = drums ahead) is stored for edge rendering. Pocket score suppressed when BPM is null (rubato).

    **Imports:**
    ```typescript
    import type { BeatState } from './types';
    ```

    **Constants:**
    ```typescript
    const POCKET_WINDOW_MS = 80;    // +-80ms cross-correlation window
    const POCKET_CAP = 8;           // rolling 8-beat average
    const STALENESS_SEC = 0.5;      // onset older than 500ms is stale
    ```

    **computeSyncScore (exported):**
    ```typescript
    export function computeSyncScore(
      drumOnsetSec: number,
      bassOnsetSec: number,
    ): { score: number; offsetMs: number } {
      const offsetMs = (drumOnsetSec - bassOnsetSec) * 1000;  // positive = drums ahead
      const absOffset = Math.abs(offsetMs);
      const score = absOffset <= POCKET_WINDOW_MS
        ? 1 - (absOffset / POCKET_WINDOW_MS)
        : 0;
      return { score, offsetMs };
    }
    ```

    **updatePocketScore (exported):**
    Called every tick. Checks for new onset pairs and updates pocket score.
    ```typescript
    export function updatePocketScore(
      beat: BeatState,
      audioTimeSec: number,
    ): void {
      // Rubato suppression (BEAT-10): suppress pocket score when BPM is null
      if (beat.bpm === null) {
        beat.pocketScore = 0;
        return;
      }

      // Check staleness: both onsets must be recent (within 500ms)
      const drumAge = beat.lastDrumOnsetSec > 0
        ? audioTimeSec - beat.lastDrumOnsetSec
        : Infinity;
      const bassAge = beat.lastBassOnsetSec > 0
        ? audioTimeSec - beat.lastBassOnsetSec
        : Infinity;

      // Only compute sync when both onsets are fresh
      if (drumAge > STALENESS_SEC || bassAge > STALENESS_SEC) {
        // No fresh pair — keep existing rolling average unchanged
        return;
      }

      // Check if this is a NEW onset pair (avoid recomputing same pair)
      // Both must be within 200ms of each other to be considered a pair
      const pairGap = Math.abs(beat.lastDrumOnsetSec - beat.lastBassOnsetSec);
      if (pairGap > 0.2) return;  // onsets too far apart to be a musical pair

      // Compute sync score
      const { score, offsetMs } = computeSyncScore(
        beat.lastDrumOnsetSec,
        beat.lastBassOnsetSec,
      );

      // Update timing offset (BEAT-09)
      beat.timingOffsetMs = offsetMs;

      // Push sync score into rolling 8-beat pocket buffer
      beat.pocketBuffer[beat.pocketHead] = score;
      beat.pocketHead = (beat.pocketHead + 1) % POCKET_CAP;
      if (beat.pocketSamples < POCKET_CAP) beat.pocketSamples++;

      // Compute rolling average
      const n = beat.pocketSamples;
      let sum = 0;
      for (let i = 0; i < n; i++) sum += beat.pocketBuffer[i];
      beat.pocketScore = sum / n;
    }
    ```
